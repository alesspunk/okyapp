export function DiscountRibbon({ tipo = 'List', type = 'Normal', small = false, label = '25% OFF' }) {
  const typeClass = `discount-ribbon-type-${type.toLowerCase().replace(/\s+/g, '-')}`;
  if (tipo === 'Wrap') {
    return (
      <div className={`discount-ribbon discount-ribbon-wrap ${small ? 'discount-ribbon-wrap-small' : ''} ${typeClass}`}>
        <span className="discount-ribbon-text token-price-percent">{label}</span>
      </div>
    );
  }
  return (
    <div className={`discount-ribbon discount-ribbon-list ${typeClass}`}>
      <span className="discount-ribbon-text token-price-percent">{label}</span>
    </div>
  );
}

Show/Hide Code Best Practice

  • Keep snippets collapsed by default to prioritize visual inspection.
  • Expose one focused snippet per component API (avoid full-file dumps).
  • Use semantic disclosure with <details> and <summary>.
  • Include typed props + defaults + key class/tokens in each example.

type CodeExampleProps = {
  title?: string;
  children: React.ReactNode;
};

export function CodeExample({ title = 'Show React snippet', children }: CodeExampleProps) {
  return (
    <details>
      <summary><strong>{title}</strong></summary>
      <pre><code>{children}</code></pre>
    </details>
  );
}
